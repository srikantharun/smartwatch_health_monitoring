API Example

import requests
import json

# Make a prediction
prediction_data = {
    "features": {
        "feature1": 1.5,
        "feature2": 2.0,
        "feature3": 3.0
    }
}

response = requests.post(
    "http://localhost:8000/api/v1/predictions/predict",
    json=prediction_data
)

print(json.dumps(response.json(), indent=2))
